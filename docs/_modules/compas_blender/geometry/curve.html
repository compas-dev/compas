    def __init__(self, object):
        self.guid = object.name
        self.curve = object
        self.geometry = None
        self.attributes = {}
        self.type = self.curve.type

    @classmethod
    def from_selection(cls):
        object = select_curve()
        return cls(object)

    @property
    def xyz(self):
        return list(self.curve.location)

        self.curve.hide = True

    def show(self):
